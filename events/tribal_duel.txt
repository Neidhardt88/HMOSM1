tribal_duel.2 = {
 	type = character_event
 	hidden = yes

 	trigger = {
 		scope:actor.current_ruler = {
 			is_alive = yes
 		}
 		scope:target = {
 			is_alive = yes
 		}
 		scope:champion = {
 			is_alive = yes
 		}
 	}

	immediate = {
		set_variable = {
			name = roll
				value = { integer_range = { min = 1 max = 20 } }
		}
		if = { 
			limit = {
				has_variable = roll
				var:roll <= scope:actor.current_ruler.martial # Ruler attacks
			}
			
			scope:actor.current_ruler.employer = { #Success
				trigger_event = { 
					id = tribal_duel.4
					days = 3
				}
			}
		}	
		else = {
			scope:actor.current_ruler.employer = { #Failure
				trigger_event = { 
					id = tribal_duel.5
					days = 3
				}
			}
		}
		remove_variable = roll
 	}

}

tribal_duel.4 = { # Ruler win
	type = country_event
	title = "tribal_duel.4.t"
	desc = "tribal_duel.4.desc"
	picture = gladiators
	left_portrait = scope:actor.current_ruler
	right_portrait = scope:target

	immediate = {
		scope:target = {
			add_health = { -15 -85 }
		}

 	}

	option = { # Let them live
 		name = "tribal_duel.4.a"

 		scope:actor.current_ruler = {
 			add_popularity = 15
 			random_list = {
 				50 = {
 					trigger = {
 						NOT = {
 							has_trait = merciful
 						}
 					}
 					add_trait = merciful
 				}
 				30 = {
 					trigger = {
 						NOT = {
 							has_trait = just
 						}
 					}
 					add_trait = just
 				}
 				20 = {
 					trigger = {
 						NOT = {
 							has_trait = forgiving
 						}
 					}
 					add_trait = forgiving
 				}
 				10 = {
 					trigger = {
 						NOT = {
 							has_trait = ambitious
 						}
 					}
 					add_trait = ambitious
 				}
 			}
 		}
 		scope:target = {
 			add_loyalty = tribal_duel_mercy
 			remove_rival = scope:actor.current_ruler
 		}
 	}

 	option = { # Finish them
 		name = "tribal_duel.4.b"

 		scope:actor = {
			add_political_influence = 15
 			add_tyranny = 15
 			add_treasury = {
 				value = scope:target.wealth
 				divide = 5
 			}
 		}
 		scope:actor.current_ruler = {
 			random_list = {
 				25 = {
 					trigger = {
 						NOT = {
 							has_trait = cruel
 						}
 					}
 					add_trait = cruel
 				}
 				25 = {
 					trigger = {
 						NOT = {
 							has_trait = harsh
 						}
 					}
 					add_trait = harsh
 				}
 				25 = {
 					trigger = {
 						NOT = {
 							has_trait = vengeful
 						}
 					}
 					add_trait = vengeful
 				}
 				25 = {
 					trigger = {
 						NOT = {
 							has_trait = ambitious
 						}
 					}
 					add_trait = ambitious
 				}
 			}
 		}
 		scope:target = {
 			death = {
 				death_reason = death_execution
 				killer = scope:actor.current_ruler
 			}
 		}
 	}
}
