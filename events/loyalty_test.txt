namespace = loyalty_test

loyalty_test.1 = { #You hit
 	type = character_event
 	hidden = yes

	immediate = {
		set_variable = {
			name = roll
				value = { integer_range = { min = 1 max = 20 } }
		}
		if = { 
			limit = {
				has_variable = roll
				var:roll <= scope:target.zeal 
			}
			
			scope:loyalty_officer = { #He hits
				trigger_event = { 
					id = loyalty_test.3
					days = 3
				}
			}
		}	
		else = {
			scope:actor = { #YOU WIN
				trigger_event = { 
					id = loyalty_test.4
					days = 3
				}
			}
		}
		remove_variable = roll
 	}
}
