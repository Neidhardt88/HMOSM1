	#interface_lock = no

	immediate = {
		scope:mission_spy = {
				trigger_event = { id = assign_mission.210 }

 	}

	}
}

assign_mission.210 = {
	type = character_event
 	hidden = yes

 	trigger = {
		
	}

	immediate = {

		set_variable = {
			name = roll
			value = { integer_range = { min = 1 max = 20 } }
		}
		if = { 
			limit = {
				has_variable = roll
				var:roll <= scope:mission_spy.charisma
			}
			scope:mission_spy = {
				trigger_event = { 
					id = assign_mission.34 
					days = 30
				}
			}
		}
		else = {
			scope:mission_origin_country = {
				trigger_event = { 
					id = assign_mission.35
					days = 30
				}
			}
		}
		remove_variable = roll
 	}	
}
