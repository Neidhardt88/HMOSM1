country_decisions = {
	
	# Form Great Thrace (Thracian)
	# Suggested and designed by Jandoski

	form_great_thrace = {
		
		potential = {
			num_of_cities >= 1
			NOT = { tag = GTR }
			OR = {
				primary_culture = grecothracian
				country_culture_group = thrace_group
			}
			NOT = {
				is_tier_2_formable_trigger = yes
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {
				OR = {
					is_in_region = thrace_region
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = pannonia_not_exists
				NOT = {
					any_country = {
						tag = GTR
					}
				}
			}

			can_form_nation_trigger = yes
			owns_or_subject_owns_region = thrace_region
		}
		
		effect = {
			custom_tooltip = "tier_2_formable_tooltip"
			change_country_name = "GREAT_THRACE_THRACIAN_NAME"
			if = {			
			capital_scope = {
				establish_city_effect = yes
				capital_formable_medium_effect = yes
				formable_capital_modifier_normal_effect = yes
			}
			hidden_effect = {
				change_country_adjective = "GREAT_THRACIAN_ADJECTIVE"
				change_country_color =  "thrace_thracian_color"
				change_country_tag = GTR
				change_country_flag = THRACE_THRACIAN_FLAG
				trigger_event = {
					id = formable_heritage.167
				}
			}
			add_5_free_province_investments = yes
		}
		
		ai_will_do = {
			base = 1
			modifier = {
				add = 10
				current_date > 600.1.1
			}
		}
	}

}
