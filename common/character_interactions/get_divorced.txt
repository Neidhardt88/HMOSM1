get_divorced = { 

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_arrange_marriage"
	
	message = no
	
	
	potential_trigger = {
		scope:actor = {
			OR = {
				is_monarchy = yes 
				is_tribal = yes
				#is_republic = yes
			}
			tyranny <= 99
		}
		hidden:scope:target = {
			is_alive = yes
			employer = scope:actor
			is_married = yes
			spouse = {
				is_alive = yes
				is_ruler = yes 
			}
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			is_married = yes
			is_adult = yes
			prisoner = no			
		}
	}
	
	
	effect = {
		scope:target = {
			divorce_character = spouse
			add_loyalty = divorced_by_ruler
			#add_prominence = -50
			hidden_effect = {
				family = {
					every_family_member = {
						limit = {
							employer = scope:actor
							is_ruler = no
						}
						add_loyalty = family_forced_to_divorce 
					}
				} 
			}
		}

		scope:actor = { 
			add_tyranny = {
				value = scope:target.power_base
				divide = 2
				min = 10
			}
		}
		scope:actor.current_ruler = { 
			add_popularity = -30
			#add_prominence = -30
		}
	}	
}
